21 void TemplateParameterSymbol::Write(SymbolWriter& writer)
22 {
23     TypeSymbol::Write(writer);
24     writer.GetBinaryWriter().Write(hasDefault);
25     if (hasDefault)
26     {
27         writer.GetBinaryWriter().Write(defaultStr);
28     }
29 }
30 
31 void TemplateParameterSymbol::Read(SymbolReader& reader)
32 {
33     TypeSymbol::Read(reader);
34     hasDefault = reader.GetBinaryReader().ReadBool();
35     if (hasDefault)
36     {
37         defaultStr = reader.GetBinaryReader().ReadUtf8String();
38     }
39 }

51 TypeSymbol* TemplateParameterSymbol::UnifyTemplateArgumentType(SymbolTable& symbolTable, const std::unordered_map<TemplateParameterSymbol*, TypeSymbol*>& templateParameterMap, const Span& span)
52 {
53     auto it = templateParameterMap.find(this);
54     if (it != templateParameterMap.cend())

68 std::unique_ptr<dom::Element> BoundTemplateParameterSymbol::CreateDomElement(TypeMap& typeMap)
69 {
70     std::unique_ptr<dom::Element> element(new dom::Element(U"BoundTemplateParameterSymbol"));
71     if (type)
72     {
73         std::unique_ptr<dom::Element> typeElement(new dom::Element(U"type"));
74         int typeId = typeMap.GetOrInsertType(type);
75         typeElement->SetAttribute(U"ref", U"type_" + ToUtf32(std::to_string(typeId)));
76         element->AppendChild(     std::unique_ptr<dom::Node>(typeElement.release()));
77     }
78     return element;
79 }
